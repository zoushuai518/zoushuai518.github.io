#### php 并发写文件|文件锁

<pre><code>
$fp = fopen('b.txt', 'w+');
if (flock($fp, LOCK_EX)) { // 进行排它锁定
	fwrite($fp, 'Write something here\n');
	flock($fp, LOCK_UN);
} else {
	echo 'Couldn't lock the file !';
}
fclose($fp);
</pre></code>
